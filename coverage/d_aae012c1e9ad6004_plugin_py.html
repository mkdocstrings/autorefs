1"""This module contains the "mkdocs-autorefs" plugin. 

2 

3After each page is processed by the Markdown converter, this plugin stores absolute URLs of every HTML anchors 

4it finds to later be able to fix unresolved references. 

5It stores them during the [`on_page_content` event hook](https://www.mkdocs.org/user-guide/plugins/#on_page_content). 

6 

7Just before writing the final HTML to the disc, during the 

8[`on_post_page` event hook](https://www.mkdocs.org/user-guide/plugins/#on_post_page), 

9this plugin searches for references of the form `[identifier][]` or `[title][identifier]` that were not resolved, 

10and fixes them using the previously stored identifier-URL mapping. 

11""" 

12 

13from __future__ import annotations 

14 

15import contextlib 

16import functools 

17import logging 

18from typing import TYPE_CHECKING, Any, Callable, Sequence 

19from urllib.parse import urlsplit

21from mkdocs.plugins import BasePlugin 

22from mkdocs.utils import warning_filter

24from mkdocs_autorefs.references import AutorefsExtension, fix_refs, relative_url

27 from mkdocs.config import Config

31log = logging.getLogger(f"mkdocs.plugins.{__name__}")

35class AutorefsPlugin(BasePlugin): 

36 """An `mkdocs` plugin. 

37 

38 This plugin defines the following event hooks: 

39 

40 - `on_config` 

41 - `on_page_content` 

42 - `on_post_page` 

43 

44 Check the [Developing Plugins](https://www.mkdocs.org/user-guide/plugins/#developing-plugins) page of `mkdocs` 

45 for more information about its plugin system. 

46 """ 

47 

48 scan_toc: bool = True 

49 current_page: str | None = None 

50 

51 def __init__(self) -> None: 

52 """Initialize the object.""" 

53 super().__init__() 

54 self._url_map: dict[str, str] = {} 

55 self._abs_url_map: dict[str, str] = {} 

56 self.get_fallback_anchor: Callable[[str], str | None] | None = None 

57 

58 def register_anchor(self, page: str, identifier: str) -> None: 

59 """Register that an anchor corresponding to an identifier was encountered when rendering the page. 

60 

61 Arguments: 

62 page: The relative URL of the current page. Examples: `'foo/bar/'`, `'foo/index.html'` 

63 identifier: The HTML anchor (without '#') as a string. 

64 """ 

65 self._url_map[identifier] = f"{page}#{identifier}" 

66 

67 def register_url(self, identifier: str, url: str) -> None: 

68 """Register that the identifier should be turned into a link to this URL. 

69 

70 Arguments: 

71 identifier: The new identifier. 

72 url: The absolute URL (including anchor, if needed) where this item can be found. 

73 """ 

74 self._abs_url_map[identifier] = url 

75 

76 def _get_item_url( 

77 self, 

78 identifier: str, 

79 fallback: Callable[[str], Sequence[str]] | None = None, 

80 ) -> str: 

81 try: 

82 return self._url_map[identifier] 

83 except KeyError: 

84 if identifier in self._abs_url_map: 

85 return self._abs_url_map[identifier] 

86 if fallback: 

87 new_identifiers = fallback(identifier) 

88 for new_identifier in new_identifiers: 

89 with contextlib.suppress(KeyError): 

90 url = self._get_item_url(new_identifier) 

91 self._url_map[identifier] = url 

92 return url 

93 raise 

94 

95 def get_item_url( 

96 self, 

97 identifier: str, 

98 from_url: str | None = None, 

99 fallback: Callable[[str], Sequence[str]] | None = None, 

100 ) -> str: 

101 """Return a site-relative URL with anchor to the identifier, if it's present anywhere. 

102 

103 Arguments: 

104 identifier: The anchor (without '#'). 

105 from_url: The URL of the base page, from which we link towards the targeted pages. 

106 fallback: An optional function to suggest alternative anchors to try on failure. 

107 

108 Returns: 

109 A site-relative URL. 

110 """ 

111 url = self._get_item_url(identifier, fallback) 

112 if from_url is not None: 

113 parsed = urlsplit(url) 

114 if not parsed.scheme and not parsed.netloc: 

115 return relative_url(from_url, url) 

116 return url

118 def on_config(self, config: Config, **kwargs: Any) -> Config: # noqa: ARG002 

119 """Instantiate our Markdown extension. 

120 

121 Hook for the [`on_config` event](https://www.mkdocs.org/user-guide/plugins/#on_config). 

122 In this hook, we instantiate our [`AutorefsExtension`][mkdocs_autorefs.references.AutorefsExtension] 

123 and add it to the list of Markdown extensions used by `mkdocs`. 

124 

125 Arguments: 

126 config: The MkDocs config object. 

127 kwargs: Additional arguments passed by MkDocs. 

128 

129 Returns: 

130 The modified config. 

131 """ 

132 log.debug(f"{__name__}: Adding AutorefsExtension to the list") 

133 config["markdown_extensions"].append(AutorefsExtension()) 

134 return config

136 def on_page_markdown(self, markdown: str, page: Page, **kwargs: Any) -> str: # noqa: ARG002 

137 """Remember which page is the current one. 

138 

139 Arguments: 

140 markdown: Input Markdown. 

141 page: The related MkDocs page instance. 

142 kwargs: Additional arguments passed by MkDocs. 

143 

144 Returns: 

145 The same Markdown. We only use this hook to map anchors to URLs. 

146 """ 

147 self.current_page = page.url 

148 return markdown 

149 

150 def on_page_content(self, html: str, page: Page, **kwargs: Any) -> str: # noqa: ARG002 

151 """Map anchors to URLs. 

152 

153 Hook for the [`on_page_content` event](https://www.mkdocs.org/user-guide/plugins/#on_page_content). 

154 In this hook, we map the IDs of every anchor found in the table of contents to the anchors absolute URLs. 

155 This mapping will be used later to fix unresolved reference of the form `[title][identifier]` or 

156 `[identifier][]`. 

157 

158 Arguments: 

159 html: HTML converted from Markdown. 

160 page: The related MkDocs page instance. 

161 kwargs: Additional arguments passed by MkDocs. 

162 

163 Returns: 

164 The same HTML. We only use this hook to map anchors to URLs. 

165 """ 

166 if self.scan_toc: 

167 log.debug(f"{__name__}: Mapping identifiers to URLs for page {page.file.src_path}") 

168 for item in page.toc.items: 

169 self.map_urls(page.url, item) 

170 return html 

171 

172 def map_urls(self, base_url: str, anchor: AnchorLink) -> None: 

173 """Recurse on every anchor to map its ID to its absolute URL. 

174 

175 This method populates `self.url_map` by side-effect. 

176 

177 Arguments: 

178 base_url: The base URL to use as a prefix for each anchor's relative URL. 

179 anchor: The anchor to process and to recurse on. 

180 """ 

181 self.register_anchor(base_url, anchor.id) 

182 for child in anchor.children: 

183 self.map_urls(base_url, child) 

184 

185 def on_post_page(self, output: str, page: Page, **kwargs: Any) -> str: # noqa: ARG002 

186 """Fix cross-references. 

187 

188 Hook for the [`on_post_page` event](https://www.mkdocs.org/user-guide/plugins/#on_post_page). 

189 In this hook, we try to fix unresolved references of the form `[title][identifier]` or `[identifier][]`. 

190 Doing that allows the user of `autorefs` to cross-reference objects in their documentation strings. 

191 It uses the native Markdown syntax so it's easy to remember and use. 

192 

193 We log a warning for each reference that we couldn't map to an URL, but try to be smart and ignore identifiers 

194 that do not look legitimate (sometimes documentation can contain strings matching 

195 our [`AUTO_REF_RE`][mkdocs_autorefs.references.AUTO_REF_RE] regular expression that did not intend to reference anything). 

196 We currently ignore references when their identifier contains a space or a slash. 

197 

198 Arguments: 

199 output: HTML converted from Markdown. 

200 page: The related MkDocs page instance. 

201 kwargs: Additional arguments passed by MkDocs. 

202 

203 Returns: 

204 Modified HTML. 

205 """ 

206 log.debug(f"{__name__}: Fixing references in page {page.file.src_path}") 

207 

208 url_mapper = functools.partial(self.get_item_url, from_url=page.url, fallback=self.get_fallback_anchor) 

209 fixed_output, unmapped = fix_refs(output, url_mapper) 

210 

211 if unmapped and log.isEnabledFor(logging.WARNING): 

212 for ref in unmapped: 

213 log.warning( 

214 f"{__name__}: {page.file.src_path}: Could not find cross-reference target '[{ref}]'", 

215 ) 

216 

217 return fixed_output
